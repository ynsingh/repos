  27  class Auth_OpenID_CryptUtil {
  28      /**
  29       * Get the specified number of random bytes.
  30       *
  31       * Attempts to use a cryptographically secure (not predictable)
  32       * source of randomness if available. If there is no high-entropy
  33       * randomness source available, it will fail. As a last resort,
  34       * for non-critical systems, define
  35       * <code>Auth_OpenID_RAND_SOURCE</code> as <code>null</code>, and
  36       * the code will fall back on a pseudo-random number generator.
  37       *
  38       * @param int $num_bytes The length of the return value
  39       * @return string $bytes random bytes
  40       */
  41      function getBytes($num_bytes)
  42      {
  43          static $f = null;
  44          $bytes = '';
  45          if ($f === null) {
  46              if (Auth_OpenID_RAND_SOURCE === null) {
  47                  $f = false;
  48              } else {
  49                  $f = @fopen(Auth_OpenID_RAND_SOURCE, "r");
  50                  if ($f === false) {
  51                      $msg = 'Define Auth_OpenID_RAND_SOURCE as null to ' .
  52                          ' continue with an insecure random number generator.';
  53                      trigger_error($msg, E_USER_ERROR);
  54                  }
  55              }
  56          }
  57          if ($f === false) {
  58              // pseudorandom used
  59              $bytes = '';
  60              for ($i = 0; $i < $num_bytes; $i += 4) {
  61                  $bytes .= pack('L', mt_rand());
  62              }
  63              $bytes = substr($bytes, 0, $num_bytes);
  64          } else {
  65              $bytes = fread($f, $num_bytes);
  66          }
  67          return $bytes;
  68      }
  69  
  70      /**
  71       * Produce a string of length random bytes, chosen from chrs.  If
  72       * $chrs is null, the resulting string may contain any characters.
  73       *
  74       * @param integer $length The length of the resulting
  75       * randomly-generated string
  76       * @param string $chrs A string of characters from which to choose
  77       * to build the new string
  78       * @return string $result A string of randomly-chosen characters
  79       * from $chrs
  80       */
  81      function randomString($length, $population = null)
  82      {
  83          if ($population === null) {
  84              return Auth_OpenID_CryptUtil::getBytes($length);
  85          }
  86  
  87          $popsize = strlen($population);
  88  
  89          if ($popsize > 256) {
  90              $msg = 'More than 256 characters supplied to ' . __FUNCTION__;
  91              trigger_error($msg, E_USER_ERROR);
  92          }
  93  
  94          $duplicate = 256 % $popsize;
  95  
  96          $str = "";
  97          for ($i = 0; $i < $length; $i++) {
  98              do {
  99                  $n = ord(Auth_OpenID_CryptUtil::getBytes(1));
 100              } while ($n < $duplicate);
 101  
 102              $n %= $popsize;
 103              $str .= $population[$n];
 104          }
 105  
 106          return $str;
 107      }
